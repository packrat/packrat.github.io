Here is a simple example of load sharing across docker containers using nginx. First we create a very simple server using go

package main

import ( "fmt" "log" "net/http" "os" )

var n int = 1 var idstr string

func HelloServer(w http.ResponseWriter, req *http.Request) { fmt.Fprintf(w, "Hello World %s %d\n", idstr, n) n = n + 1 }

func main() { idstr = os.Args[1] http.HandleFunc("/hello", HelloServer) err := http.ListenAndServe(":12345", nil) if err != nil { log.Fatal("ListenAndServe: ", err) } }
